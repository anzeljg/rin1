def PoisciRime(besede):
  h = {} # Prazen slovar.
  for s in besede:
    # Določimo končnico trenutne besede s.
    i = len(s) - 1
    while i > 0 and s[i] not in "aeiou": i -= 1
    koncnica = s[i:]

    # Če je še ni v slovarju, jo dodajmo.
    if koncnica not in h:
      h[koncnica] = []

    # Dodajmo besedo v seznam, ki pripada tej končnici.
    h[koncnica].append(s)

  # Izpišimo rezultate.
  for koncnica in h:
    print(koncnica + ": " + str(h[koncnica]))

  # Poglejmo, koliko skupin kakšne velikosti je nastalo.
  stSkupin = {}
  maxVelikost = 0; maxSkupina = None
  for koncnica in h:
    skupina = h[koncnica]
    velikost = len(skupina)
    stSkupin[velikost] = stSkupin.get(velikost, 0) + 1
    if velikost > maxVelikost:
      maxVelikost = velikost
      maxSkupina = skupina

  # Izpišimo rezultate.
  for velikost in stSkupin:
    print(str(stSkupin[velikost]) + " skupin velikosti " +
          str(velikost))
  print("Največja skupina (" + str(maxVelikost) + " besed): " +
        maxSkupina)

PoisciRime(["zob", "miš", "rob", "grob", "piš"])

Primeri uporabe slovarja

Vaja

Skupine rimajočih se besed, ki jih dobimo pri funkciji na prejšnji strani, so lahko seveda različno velike. Dopolni funkcijo tako, da izpiše statistike o tem, koliko skupin posamezne velikosti je nastalo in katera je največja (torej katera vsebuje največ besed).

Vaja

Poišči kakšno daljše besedilo in napiši program, ki ga prebere, razbije na besede (pri tem pobriše ločila, spremeni velike črke v male ipd.) in potem na seznamu vseh (različnih) besed iz tega besedila požene postopek iz prejšnje strani, za odkrivanje skupin rimajočih se besed. Katera tako dobljena skupina je največja?
