def Obrni(slovar):
  obrnjeni = {}
  for kljuc in slovar:
    vrednost = slovar[kljuc]
    obrnjeni[vrednost] = kljuc
  return obrnjeni

# Kaj se zgodi, če ima v vhodnem slovarju več ključev enako vrednost?
# V obrnjenem slovarju bo ta vrednost seveda nastopila (kot ključ) le
# enkrat, njena pripadajoča vrednost pa bo zadnji ključ iz vhodnega
# slovarja, ki je imel to pripadajočo vrednost. "Zadnji" je mišljeno
# glede na vrstni red, v katerem zanka "for kljuc in slovar" pregleduje
# ključe. Glede tega, kakšen točno je ta vrstni red, pa nam jezik
# python ne daje nobenih posebnih zagotovil.

print(Obrni({"ena": 11, "dve": 222}))

def Komp(h1, h2):
  rezultat = {}
  for k in h1:
    k2 = h1[k]
    if k2 in h2:
      rezultat[k] = h2[k2]
  return rezultat

h1 = {"ena": 1, "dve": 2, "tri": 3}
h2 = {1: "ane", 3: "irt"}
print(Komp(h1, h2))
# izpiše {"ena": "ane", "tri": "irt"}

def Komp(h1, h2):
  rezultat = {}
  for k in h1:
    k2 = h1[k]
    if k2 in h2:
      rezultat[k] = h2[k2]
  return rezultat

def Komp3(h1, h2, h3):
  return Komp(h1, Komp(h2, h3))

# Druga različica rešitve...
def Komp3_druga(h1, h2, h3):
  h = {}
  for k in h1:
    v = h1[k]
    if v not in h2: continue
    v = h2[v]
    if v not in h3: continue
    h[k] = h3[v]
  return h

h1 = {"ena": 1, "dve": 2, "tri": 3}
h2 = {1: "ane", 3: "irt"}
h3 = {"ane": "one", "irt": "three"}
print(Komp3(h1, h2, h3))
# izpiše {"ena": "one", "tri": "three"}
print(Komp3_druga(h1, h2, h3))
# izpiše {"ena": "one", "tri": "three"}

Rešitev

Slovarja Komp(Komp(h1, h2), h3) in Komp(h1, Komp(h2, h3)) sta res vedno enaka. Še več, sta tudi enaka slovarju Komp3(h1, h2, h3). Dokaz, da je to res, spada bolj v matematiko kot v računalništvo, pa si ga vseeno oglejmo.

  • Označimo h12 = Komp(h1, h2), h23 = Komp(h2, h3), L = Komp(h12, h3), D = Komp(h1, h23). Naloga sprašuje, ali je L == D (za vse možne h1, h2, h3).
  • Recimo, da je nek ključ k prisoten v L z vrednostjo v.
    (1) To pomeni, da je k prisoten v h12, da je h12[k] prisotna v h3 in da je h3[h12[k]] == v.
    (2) Iz dejstva, da je k prisoten v h12, sledi tudi, da je k prisoten v h1, da je h1[k] prisotna v h2 in da je h2[h1[k]] == h12[k].
    (3) Označimo u = h1[k]; vidimo torej, da je u prisoten v h2 in da je h2[u] = h12[k].
    (4) Iz h3[h12[k]] == v (kar smo videli v (1)) lahko zdaj zaključimo, da je h3[h2[u]] == v.
    (5) Iz (3) in (4) zdaj dobimo, da je u prisoten v h2, da je h2[u] prisoten v h3 in da je h3[h2[u]] == v, torej je u prisoten kot ključ tudi v h23 in to s pripadajočo vrednostjo v.
    (6) Ker je torej k prisoten v h1 in je h1[k] (kar je enako u) prisotna kot ključ v h23 z vrednostjo v, mora biti k prisoten kot ključ tudi v Komp(h1, h23) z vrednostjo v, torej je D[k] == v. Tako torej vidimo, da je vsak ključ, ki je prisoten v L, prisoten tudi v D in to z enako pripadajočo vrednostjo.
  • Dokaz v obratno smer (vsak ključ, ki je prisoten v D, je prisoten tudi v L in to z enako pripadajočo vrednostjo) gre na podoben način in ti ga prepustimo za vajo.
  • Oboje skupaj nam pove, da sta slovarja L in D res enaka.

def VerigaSlovarjev(t):
  rezultat = {}
  prvi = t[0]; ostali = t[1:]
  for k in prvi:
    ok = True
    v = prvi[k]
    for slovar in ostali:
      if v not in slovar:
        ok = False; break
      v = slovar[v]
    if ok:
      rezultat[k] = v
  return rezultat

def VerigaSlovarjev_druga(t):
  rezultat = t[0]
  for slovar in t[1:]:
    rezultat = Komp(rezultat, slovar)
  return rezultat

# Predpostavimo, da v besedilu ni ločil in podobnih stvari, tako da ga
# lahko razbijemo na besede kar s funkcijo split().
def Konkordanca(besedilo):
  # Predelajmo besedilo iz niza v seznam besed.
  besedilo = besedilo.lower().split()
  n = len(besedilo)

  # Pripravimo si slovar, v katerem je za vsako besedo po en ključ,
  # pripadajoča vrednost pa je seznam pojavitev (indeksov v tabelo
  # "besede").
  pojavitve = {}
  for i in range(n):
    beseda = besedilo[i]
    if beseda not in pojavitve:
      pojavitve[beseda] = []
    pojavitve[beseda].append(i)

  # Pripravimo si urejen seznam besed.
  besede = [beseda for beseda in pojavitve]
  besede.sort()

  # Izpišimo konkordanco.
  kontekst = 5
  for beseda in besede:
    poj = pojavitve[beseda]
    # Za vsako pojavitev sestavimo levi in desni kontekst, torej
    # prejšnjih in naslednjih nekaj besed. Med levimi konteksti si
    # zapomnimo dolžino najdaljšega.
    levi = []; desni = []
    maxLevi = 0
    for i in poj:
      leviKontekst = besedilo[max(i - kontekst, 0) : i]
      desniKontekst = besedilo[i + 1 : i + 1 + kontekst]
      leviKontekst = " ".join(leviKontekst)
      desniKontekst = " ".join(desniKontekst)
      maxLevi = max(maxLevi, len(leviKontekst))
      levi.append(leviKontekst)
      desni.append(desniKontekst)
    print(beseda + " (" + str(len(poj)) + " pojavitev):")

    # Zdaj lahko vse pojavitve skupaj s konteksti izpišemo, pri čemer
    # jih z vrivanjem presledkov na začetku poravnajmo tako, da
    # opazovana beseda pride v vsaki vrstici na isto mesto. Tako je
    # izpis preglednejši.
    for i in range(len(poj)):
      leviKontekst = levi[i]
      desniKontekst = desni[i]
      print(" " * (maxLevi - len(leviKontekst) + 5) + leviKontekst +
            " | " + beseda + " | " + desniKontekst)

            in enake pravice obdarjeni so z razumom in vestjo in bi
            morali ravnati drug z drugim kakor bratje")
