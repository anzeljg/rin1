Rešitev

Prilagamo kodo rešitve, medtem ko sama razlaga rešitve presega raven tega učbenika:

import sys

def sign(x):
  return 1 if x > 0 else (-1 if x < 0 else 0)

def integral(pot, l, d):
  """
  Ploščina pod lomljenko pot od x = l do x = d.
  """
  def get_y(x_1, y_1, x_2, y_2, x):
    k = (y_2 - y_1) / (x_2 - x_1)
    return y_1 + k * (x - x_1)
  l, d = min(l, d), max(l, d)
  pl = 0
  for i in range(len(pot)-1):
    x_1, y_1 = pot[i]
    x_2, y_2 = pot[i+1]
    if x_2 <= l: continue
    if d <= x_1: continue
    x_l = max(x_1, l)
    x_d = min(x_2, d)
    pl += (x_d - x_l) * get_y(x_1, y_1, x_2, y_2, (x_l + x_d) / 2)
  return pl

n, p, q = map(int, sys.stdin.readline().split())
razmerje = p / q
grunt = [tuple(map(int, sys.stdin.readline().split())) for i in range(n)]
x_min = min(x for x, _ in grunt)
x_max = max(x for x, _ in grunt)
grunt.append(grunt[0])
odsek = {-1: [], 0: [], 1: []}
for i in range(n):
  u, v = grunt[i:i+2]
  smer = sign(v[0] - u[0])
  odsek[smer].append(u)
  odsek[smer].append(v)
zgoraj = sorted(set(odsek[1]))
spodaj = sorted(set(odsek[-1]))
if integral(spodaj, x_min, x_max) > integral(zgoraj, x_min, x_max):
  spodaj, zgoraj = zgoraj, spodaj
# Bisekcija
eps = 1e-5
a = x_min
b = x_max
while b - a > eps:
  c = (a + b) / 2
  zahod = integral(zgoraj, x_min, c) - integral(spodaj, x_min, c)
  vzhod = integral(zgoraj, c, x_max) - integral(spodaj, c, x_max)
  if zahod / vzhod < razmerje:
    a = c
  else:
    b = c
print(c)
