Rešitev

Prilagamo kodo rešitve v programskem jeziku C++, medtem ko sama razlaga rešitve presega raven tega učbenika:

#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <map>
#include <cassert>
using namespace std;

typedef struct {
  int a; int b; int c;
} triple;

bool operator < (const triple &t1, const triple &t2) {
  if (t1.a == t2.a) {
    if (t1.b == t2.b) return t1.c < t2.c;
    else return t1.b < t2.b;
  } else return t1.a < t2.a;
}

#define P2 20
#define MAXN 1 << P2

int P[P2+1][MAXN];
int N, k, cnt;
triple L[MAXN];
int id2pos[MAXN], pos2id[MAXN];

void suffixArray(const string &s) {
  N = s.size();
  for (int i = 0; i < N; i++) P[0][i] = (int)s[i];
  for (k = 1, cnt = 1; cnt < N; k++, cnt *= 2) {
    for (int i = 0; i < N; i++) { L[i].a = P[k-1][i]; L[i].b = (i+cnt < N) ? P[k-1][i+cnt]:-1; L[i].c = i; }
    sort(L, L+N);
    for (int i = 0; i < N; i++) {
      if (i != 0 && L[i].a == L[i-1].a && L[i].b == L[i-1].b) P[k][L[i].c] = P[k][L[i-1].c];
      else P[k][L[i].c] = i;
    }
  }
  for (int i = 0; i < N; i++) {
    pos2id[i] = L[i].c;
    id2pos[L[i].c] = i;
  }
}

int lcp(int x, int y) {
  int r = 0;
  if (x == y) return N-x;
  for (int len = k-1, size = 1 << len; len >= 0 && x < N && y < N; len--, size /= 2) {
    if (P[len][x] == P[len][y]) { r += size; x += size; y += size; }
  }
  return r;
}

int id2gene[MAXN];

int main() {
  int ng;
  cin >> ng;
  assert(2 <= ng);
  string s, si;
  int tot = 0;
  for (int i = 0; i < ng; i++) {
    cin >> si;
    assert(3 <= si.size());
    tot += si.size();
    if (i != 0) s += '#';
    for (int j = 0; j < si.size(); j++) id2gene[s.size()+j] = i;
    s += si;
  }
  assert(tot <= 500000);
  suffixArray(s);
  int p = 0;
  long long r = 0;
  map<int,int> f;
  for (int i = 0, j = 0; j < s.size(); j++) {
    f[id2gene[pos2id[j]]]++;
    while (f[id2gene[pos2id[i]]] > 1) {
      f[id2gene[pos2id[i]]]--;
      i++;
    }
    if (f.size() == ng) {
      int l = lcp(pos2id[i], pos2id[j]);
      int d = l - p;
      if (d > 0) r += d;
      p = l;
    }
  }
  cout << r << endl;
  return 0;
}

Biologi so našli več živalskih vrst, ki imajo zanimivo lastnost – v temi se svetijo! Upajo, da bodo z odkritjem odgovornega gena za vedno rešili problem skrivanja hišnih ljubljenčkov (npr. hrčkov pod posteljo). Določili so genske zapise vseh živali, sedaj pa jih zanima, kateri deli zapisa oz. geni so skupni vsem živalim in so zato kandidati za nadaljnje raziskovanje.
