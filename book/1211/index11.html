Implementacija s tabelo elementov

Naloga

Ponovno napiši funkcijo PrestejSamoglasnike(beseda), ki vrne število samoglasnikov v podani besedi, sestavljeni iz samih malih črk. Vendar pa si tokrat pomagaj izključno z operacijami iz modula mnozicaTabelaElementov.py. Spomni se: modul uvoziš s stavkom

import mnozicaTabelaElementov as mte

nato pa njegove funkcije kličeš kot mte.Ustvari(), mte.JePrisoten(...) itd.

import mnozicaTabelaElementov as mte

# Vrne število samoglasnikov v podanem nizu.
def PrestejSamoglasnike(niz):
samoglasniki = mte.Ustvari()
for crka in ['a', 'e', 'i', 'o', 'u']:
mte.Dodaj(samoglasniki, crka)

stevilo = 0

# Sprehodimo se po znakih niza in za vsak
# znak preverimo, ali je prisoten v množici
# samoglasnikov. Če je, števcu prištejemo 1.
for znak in niz:
if mte.JePrisoten(samoglasniki, znak):
stevilo += 1
return stevilo

print(PrestejSamoglasnike('meteorologija'))
print(PrestejSamoglasnike('vzbrst'))

Naloga

V modulu mnozicaTabelaElementov implementiraj in preizkusi še operacije Presek(m1, m2), Unija(m1, m2) in Razlika(m1, m2), ki vrnejo presek, unijo oziroma razliko podanih množic.

# Vrne presek podanih množic.
def Presek(m1, m2):
presek = []
# Za vsak element množice m1 preverimo, ali
# obstaja tudi v množici m2. Če obstaja, ga
# dodamo v presek.
for x in m1:
if JePrisoten(m2, x):
presek.append(x)
return presek

# Vrne unijo podanih množic.
def Unija(m1, m2):
# Unijo ustvarimo kot kopijo množice m1.
unija = m1[:]
# Vsak element množice m2, ki NE obstaja v
# množici m1, dodamo v unijo.
for x in m2:
if not JePrisoten(m1, x):
unija.append(x)
return unija

# Vrne razliko podanih množic.
def Razlika(m1, m2):
razlika = []
# Za vsak element množice m1 preverimo, ali
# obstaja tudi v množici m2. Če NE obstaja,
# ga dodamo v razliko.
for x in m1:
if not JePrisoten(m2, x):
razlika.append(x)
return razlika

Naloga

Časovno učinkovitost operacij lahko nekoliko izboljšamo tako, da elemente v tabeli hranimo v naraščajočem vrstnem redu. Za dodajanje in odstranjevanje potem še vedno potrebujemo čas O(n), preverjanje prisotnosti pa lahko opravimo v času O(log2n). Napiši in preizkusi modul, ki množico implementira na opisani način. Implementiraj samo operacije Ustvari, JePrisoten, Dodaj, Odstrani in NizElementov.

Spomni se na bisekcijo.

# Implementacija množice z naraščajočo
# tabelo elementov

# Vrne prazno množico.
def Ustvari():
return []

# Vrne True, če je podani element prisoten
# v podani množici, sicer pa vrne False.
def JePrisoten(mnozica, element):
# Pomagamo si s pomožno funkcijo, ki poišče
# podani element v tabeli. Če je rezultat
# funkcije večji ali enak 0, potem je
# element prisoten v tabeli, in obratno.
return (_Poisci(mnozica, element) >= 0)

# Doda podani element v podano množico.
def Dodaj(mnozica, element):
# S pomočjo funkcije _Poisci poiščemo mesto,
# kamor naj se element shrani v tabelo.
indeks = _Poisci(mnozica, element)
if indeks < 0:
mnozica.insert(-indeks - 1, element)

# Odstrani podani element iz podane množice.
def Odstrani(mnozica, element):
indeks = _Poisci(mnozica, element)
if indeks >= 0:
del mnozica[indeks]

# Če podani element v podani naraščajoče
# urejeni tabeli obstaja, vrne njegov indeks.
# V nasprotnem primeru vrne vrednost
# (-mesto - 1), kjer je 'mesto' indeks prvega
# elementa v tabeli, ki je večji od podanega
# elementa (oziroma len(tabela), če je podani
# element večji od vseh elementov tabele.
# Funkcija uporablja bisekcijo.
def _Poisci(tabela, element):
levo = 0
desno = len(tabela) - 1
while levo <= desno:
sredina = (levo + desno) // 2
if tabela[sredina] == element:
return sredina
elif element < tabela[sredina]:
desno = sredina - 1
else:
levo = sredina + 1
return -levo - 1

# Vrne niz, ki predstavlja podano množico;
# npr. '{1, 2, 3}' za množico, ki vsebuje
# elemente 1, 2 in 3.
def NizElementov(mnozica):
# Med sprehodom po tabeli elemente
# pretvarjamo v nize in nastale nize
# dodajamo v tabelo.
izpis = []
for element in mnozica:
izpis.append(str(element))

# Združimo nize iz tabele
# (npr. ['a', 'b', 'c'] --> '{a, b, c}').
return ('{' + ', '.join(izpis) + '}')
