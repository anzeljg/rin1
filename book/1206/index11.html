def Sahovnica(n, a):
  delPolja = ["#" * a, "." * a]
  vrstici = ["".join([delPolja[(x + y) % 2]
             for x in range(n)]) for y in range(2)]
  for y in range(n * a):
    print(vrstici[(y // a) % 2])

Sahovnica(3, 2)

def Predelaj(s):
  return "".join([c.lower() for c in s if c.isalpha()])

def Primerjaj(s, t):
  return Predelaj(s) < Predelaj(t)

print(Primerjaj("ab cd", "a bZd"))

def Predelaj2(s):
  vrstniRed = "abcčdefghijklmnopqrsštuvwxyzž"
  L = []
  for c in s:
    # .lower() deluje tudi na šumnikih
    c = c.lower()
    i = vrstniRed.find(c)
    if i >= 0:
      L.append(chr(i + 1))
  return "".join(L)

def Primerjaj2(s, t):
  return Predelaj2(s) < Predelaj2(t)

print(Primerjaj2("ab Šd", "abz D"))
print(Primerjaj2("abŠd", "abč"))

def CikCak(s):
  # Vpeljimo koordinatni sistem: y = 0 naj bo višina točke, v kateri se
  # naša cikcakasta črta začne; y-os kaže navzgor. Določimo najvišjo in
  # najnižjo y-koordinato, ki jo črta doseže.
  y = 0
  yMin = 0
  yMax = 0
  for c in s:
    if c == '/':
      y += 1
    else:
      y -= 1
    if y < yMin:
      yMin = y
    if y > yMax:
      yMax = y
  # V vsaki iteraciji naslednje zanke nam y0 pove, da izpisujemo vrstico,
  # ki pokriva interval y-koordinat [y0, y0+1). To pomeni, da moramo
  # izpisati vrstice od y0 = yMax-1 do vključno y0 = yMin.
  # [Vrstice y0 = yMax ne izpisujemo, ker bi bila popolnoma prazna.]
  for y0 in range(yMax-1, yMin-1, -1):
    vrstica = []
    y = 0
    for c in s:
      # Spomnimo, da nam y0 pove y-koordinato na spodnjem robu trenutne
      # vrstice. Če je c == '\\', pomeni, da je y trenutna koordinata
      # zgornjega roba vrstice, zato bomo y najprej zmanjšali in ga
      # šele nato primerjali z y0. Če pa je c == '/', to pomeni, da je
      # y že zdaj koordinata spodnjega roba vrstice, zato ga moramo
      # najprej primerjati z y in šele nato povečati.
      if c == '\\':
        y -= 1
      if y == y0:
        vrstica.append(c)
      else:
        vrstica.append('.')
      if c == '/':
        y += 1
    print("".join(vrstica))

CikCak("\\///\\/\\\\/")

Napiši funkcijo Primerjaj(s, t), ki vrne True, če pride niz s pred nizom t v abecednem vrstnem redu (ne leksikografskem), sicer pa naj vrne False. Za namen te primerjave torej ne razlikuj med velikimi črkami in malimi, poleg tega pa ignoriraj vse ne-črkovne znake v danih nizih. Predpostavi, da se v nizih s in t ne pojavljajo druge črke kot tiste iz angleške abecede (lahko velike in/ali male). Primer: Primerjaj("ab cd", "a bZd") naj vrne True.
